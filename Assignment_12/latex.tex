\section{Problem}
A continuous random variable X has a probability density function $f(x) = e^{- x}, 0<x< \infty $. Then $P\brak{X>1}$ is
\begin{enumerate}
    \item 0.368
    \item 0.5
    \item 0.632
    \item 1.0
\end{enumerate}
\section{Solution}
\begin{align}
    F_X \brak{x} &= P\brak{X \le x}\\
    &= \int_0^{x} e^{-t} dt\\
    &= 1- e^{-x}
\end{align}
Thus, the CDF of $X$ is:
\begin{align}
    F_X \brak{x} &= P\brak{X \le x}\\
    & = \begin{cases}
      0, & \text{if}\ x\le 0 \\
      1- e^{- x},  & \text{if}\ 0<x<1 \\
      1 & \text{if}\ x\ge 1
    \end{cases}
\end{align}
\begin{align}
    P\brak{X>1} &= 1-P\brak{X\le1}\\
    &= 1- \{1- e^{- 1}\}\\
    &= e^{-1}\\
    &= 0.3678 \\
    &= 0.368
\end{align}
Hence, option\brak 1 is the correct answer.\\
