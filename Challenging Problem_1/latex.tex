\section{Problem}
Let $X$ be a random variable such that $E\brak X = E\brak {X^2} = 1 $. Then, $E\brak{X^{100}} = $ ?
\begin{enumerate}
    \item 0
    \item 1
    \item $2^{100}$
    \item $2^{100} +1 $
\end{enumerate}
\section{Solution}
Given,
\begin{align}
    &E\brak X = 1\\
    \text{and }&E \brak{X^2}=1\\
    \therefore &Var\brak X = E\brak{X^2}- E^2\brak{X}=1-1=0
\end{align}
$\therefore X$ is a degenerate random variable degenerated at $X=1$.
\begin{align}
    \because & Var\brak{X}=0\\
    \implies & E\brak{X-E\brak{X}}^2 =0\\
    \implies & X= E\brak X = 1, \text{ with probability } 1.\\
    \implies & P\brak{X=1} = 1\\[1em]
    \therefore E\brak{X^{100}} &= k^{100}\times P\brak{X=k}, k=1\\
    & = 1 \times P\brak{X=1}\\
    & = 1
\end{align}
Hence, option \brak 2 is the correct answer.
